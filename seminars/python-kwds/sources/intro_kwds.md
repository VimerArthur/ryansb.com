# what are keywords?

* language builtins

# finding keywords

	!python
	>>> import keyword
	>>> keyword.kwlist
	['and', 'as', 'assert', 'break', 'class', 'continue', 'def', 'del', 'elif',
	'else', 'except', 'exec', 'finally', 'for', 'from', 'global', 'if', 'import',
	'in', 'is', 'lambda', 'not', 'or', 'pass', 'print', 'raise', 'return', 'try',
	'while', 'with', 'yield']
