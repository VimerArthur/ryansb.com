scoping and destroying

global

 1 def fu():
 2     print y
 3 
 4 def func():
 5     global y
 6     print y
 7 
 8 fu() # raises a NameError
 9 y = "hello"
10 func() # works, because global

del

1 >>> y = "Hello"
2 >>> z = object()
3 >>> print y, 'and', z
4 hello and <object object at 0xb76e4740>
5 >>> del y, z
6 >>> print y, 'and', z
7 NameError: name 'y' is not defined
